https://www.acmicpc.net/problem/2309
```C++
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int arr[9]{};
int sum{};

void Solve()

{
    for (int i = 0; i < 9; i++)
    {
        for (int j = i+1; j < 9; j++)
        {
            if(sum - arr[i] - arr[j] == 100)
            {
                for (int k = 0; k < 9; k++)
                    if(i != k && j != k)
                        cout << arr[k] << endl;
                        
                return;
            }
        }
    }
}

int main()
{
	for (int i = 0; i < 9; i++)
    {
        cin>>arr[i];
        sum+=arr[i];    
    }  
    sort(arr, arr+9);
    Solve();
}
```
- 강의에 나온대로 풀면 이상하게 정답이 아니란다
- 하지만 강의에서 두번째 풀이를 이용해 알고리즘을 적용하면 답이 나온다
- 풀이는 입력받은 9명의 키를 모두 sum에 더하여 저장해준 뒤 9명 중 2명을 뽑아 sum에서 그 키들을 빼준다
- 이때 sum이 100이면 그 두 명을 제외한 난쟁이의 키를 출력해주면 되는 문제이다
